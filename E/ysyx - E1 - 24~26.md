```c
#include <stdio.h>

// 函数原型应当是char *shrink_space(char *dest, const char *src, size_t n);
// 实现的效果是将src的字符串选取n个, 进行空格压缩并填入dest, 若字符串不满n个则后续填\0

char *shrink_space(char *dest, const char *src, size_t n)
{
    char *ret = dest;
    int flag = 0; // 表示是否已经出现空位;
    for (int count = 0; count < n; count++)
    {
        if (*src == ' '  ||
            *src == '\t' ||
            *src == '\n' ||
            *src == '\r'){
            if(!flag){
            *dest = ' ';
            dest++;
            flag = 1;}
            }
        else{
            *dest = *src;
            dest++;
            flag = 0;
            }

        if (*src != '\0')
            src++;
    }
    
    return ret;
}

int main(void)
{
    char a[30];
    printf("%s", shrink_space(a, "Hello \n \t Innoy", 15));
};
```

```c
/*尝试使用泛型的二分查找, 这里求助了AI作为参考, 返回这里返回了字符串的地址, 修改前面的cmp可以变成查找int, 返回int的地址, 总之返回的是数组中的地址位置*/
void *binary_search(
    void *base,
    size_t n,
    size_t size,
    void *key,
    int (*cmp)(const void *, const void *))
{
    char *arr = base;
    size_t left = 0, right = n;

    while (left < right)
    {
        size_t mid = (left + right) / 2;
        void *elem = arr + mid * size;

        int c = cmp(key, elem);

        if (c == 0)
            return elem;
        else if (c < 0)
            right = mid;
        else
            left = mid + 1;
    }

    return NULL;
}
```

`va_start`建立了后续用于`va_arg`和`va_end`的序列流程初始化, 必须被第一个被调用. last是用于定位参数序列读取起点的参数, 它的实参填写可变函数中想要开始序列读取的参数名就可以了.

由于`va_start()`会读取地址, 因此`last`不能被定义为`rigister`类型, 这是一种要求变量保存在寄存器的类型声明, 寄存器不能被读地址因此当然也就不能用在这里. 

`last`也不能被传递函数类型, 这是因为函数被传递时会被退化(decay)为`void (*func)()`, 这样变成指针类型的话计算内存大小就可能会出问题. 同理`last`也不能被传递 `array` 类型, 毕竟数组被传递时也会decay为指针类型. 这样的话算size就变成了指针的大小而不是数组的大小. 可以想象往后读参数的时候可能把数组内容当成参数. 

> 实际上现代GCC已经没有这些限制, 因为现代GCC中`va_start`已经是编译器內建指令, 因此清楚ABI规则.

超绝长难句, 一看一个不吱声：`The argument type is a type name specified so that the type of a pointer to an object that has the specified type can be obtained simply by adding a * to type.` 

这么绕的原因在于其函数原型中, 它有两个参数, 除了ap之外, 另一个参数的名字就叫`type`. 这样一看至少不会直接被绕晕了, 意思其实就是:

不满足条件的是谁? 显然至少函数类型是其中之一. 


注意这里的`random errors will occur` 其实就是 `undefined behavior`, 这里提醒了我们需要注意参数自动类型提升(promotion)的问题, 比如你传一个float, 可变参数是会自动将其提升为double的, 所以应该写的是double, 尤其注意char会被提升为int. 而后面提到的是跨函数问题, 其实这是理所当然的事情: va_list如果被传进别的函数并且被推进了, 那当然被修改后返回的就和原函数传入的状态不一样, 就像被NTR了一样. 

最后再看看`va_end`

```
 va_end()

```

这就简单了, 这个就是用来结束这个函数里的va_start(), 没什么特别的. 为什么va_start之后必须要va_end? 直接等到下一个start自动结束不行么. 这是因为这一套属于状态机模型, 状态机模型没有结束是不完整的. 而且va_start后, va_list会持续占用资源.

好了, 我们也是独自把man函数读完了, 现在我们可以开始理解这一段代码的行为了: 

```c
#include <stdio.h>
#include <stdarg.h>

void myprintf(const char *format, ...)
{
     va_list ap;
     char c;

     va_start(ap, format);
     while (c = *format++) {
	  switch(c) {
	  case 'c': {
	       /* char is promoted to int when passed through '...' */
	       char ch = va_arg(ap, int);
	       putchar(ch);
	       break;
	  }
	  case 's': {
	       char *p = va_arg(ap, char *);
	       fputs(p, stdout);
	       break;
	  }
	  default:
	       putchar(c);
	  }
     }
     va_end(ap);
}

int main(void)
{
     myprintf("c\ts\n", '1', "hello");
     return 0;
}
```

建立va_list ap后，可见myprintf通过一个while循环遍历输出字符串format, 遇到`\0`则while循环条件终止自动结束函数. 值得注意的是这里并没有对`\n`等函数进行判断, 其原因在前文也有过说明, 因为转义字符是在编译阶段被处理的, 这里自然就不需要对转义字符进行处理, 也处理不了. 但是这里会对转译字符进行处理, 因为转译字符的本质是对参数的处理.

我们这里梳理一下, 我们的PC从程序到cpu解析计算的整个流程:

1. 程序文件最开始保存在磁盘, 也就是外存, 内容并不被关心
2. 当你运行文件, 操作系统首先开始工作. 首先是建立进程, 然后操作系统读取程序文件的文件头, 获得程序所需要的静态资源量, 建立进程的虚拟内存模型. 
3. 装载. 程序的一部分被加载进进程所在的虚拟内存. 运行过程中代码被放入.text, 全局变量存入.data, 未初始化变量存入.bss, 动态内存申请在heap, 函数栈帧放在stack. 程序不会一次性加载全部程序代码, 通常会等待触发缺页异常(Page fault), 这是一种同步中断. 之后程序通过页表查询所需加载的磁盘页, 内核通过VMA找到对应文件加载新的磁盘页到内存. 
4. 进入函数, 也就会在stack建立frame, 这里保存了必要的局部函数域数据, (但是注意: 函数指令本身被保存的位置并没有变化), 指令流流入cpu, 运用frame内的数据进行计算, 结束之后会通过返回地址回到原先的下一条指令位置, 原先的frame被销毁. 
5. 程序运行完毕, 进程退出, 内存回收.

现在就来拆解这段宏定义: 

```c
#define va_arg(ap, T) \
	(* (T *)(((ap) += _Bnd(T, 3U)) - _Bnd(T, 3U)))
```

显然这里是个宏函数, 有`ap`, `T`两个参数, 第一个就是所谓的va_list类型的指针了, 第二个参数我们也分析过这是表示类型.

`_Bnd(X, bnd)`将类型或变量`X`的长度对齐到`bnd+1`字节的整数倍，例如`_Bnd(char, 3U)`的值是4. 

这里\_Bnd是用来提升参数的, 也就是做对齐. 注意这里的side effect: ap通过+=\_Bnd(T, 3U), 它的值往后走了一个size, 然后通过-\_Bnd()让表达式返回了之前的值. 这里ap就实现了往后遍历

(T \*) 是强制类型转换, 将ap指针被解释的方式强制转换为T type. 最后前面再加个\* 表示将值解引用, 返回地址上写的值, 在这个va_arg函数里, 实际上要取得的就是参数的实参. 

综合起来看, 这个va_arg完成了它的工作: 每次被调用都会返回当前ap指向的参数值, 然后ap往后走到下一个参数, 这里的size会自动进行类型提升.

```c
#define va_start(ap, A) \
	(void)((ap) = (char *)&(A) + _Bnd(A, 3U))
```

va_start看起来好看很多, 就是定义了ap, 类型暂时用void, 毕竟函数可以返回赋值给一个va_list类型的ap自动进行类型转换. 参数A在man里说明是last, 表达参数起始点, 这里是对A取了地址, 类型被强制转化为char\*类型后加一个size, 这样ap就指向到了last往后一个参数位置. 

>注意_Bnd是先被调用再被定义, 这在宏定义中是可以的, 毕竟只要留个记录, 日后拿来替换就行. 

>这里强制转化为char \*类型是非常有必要的, 因为指针加减运算默认是以sizeof(\*pointer)为单位的, 如果不转化的话, 那每次加的步长就可能不是1 byte. 这里费这么大劲是为了实现参数提升.

```c
#define _Bnd(X, bnd) (sizeof (X) + (bnd) & ~(bnd))
```

这里就是数学方法了, \_Bnd返回X type的大小对于数字bnd+1的整数倍. 注意运算顺序先是sizeof(X) + bnd, 然后再和~bnd进行与运算.  

至于为什么这里设计成bnd+1的倍数, 而不是bnd的倍数, 这里没有什么巧妙细节, 单纯是为了运算, 实际上这根本不是个适合人用的好用接口.

```c
# include<stdio.h>
# include<stdarg.h>

void printf_int(int num){
    if (num == 0){
        fputc('0', stdout);
        return;
    }

    if (num < 0){
        fputc('-', stdout);
        num = -num;
    }

    char buf[32] = {};
    int i = 0;
    while(num){
        buf[i++] = (char) (num%10 + '0');
        num/=10;
    }

    while(--i >= 0)
        fputc(buf[i], stdout);
}


void printf_double(double num)
{
    if (num < 0)
    {
        fputc('-', stdout);
        num = -num;
    }

    printf_int((int)num);
    fputc('.', stdout);

    for (int i=0; i<6; i++){
        num -= (int)num;
        num *= 10;
        fputc((char) ((int)num + '0'), stdout);
    }
}


void myprintf(const char *format, ...){
    char p;
    va_list ap;
    
    va_start(ap, format);

    while (*format){
        if (*format != '%'){
            fputc(*format++, stdout);
            continue;
        }

        format++;
        switch(*format){
            case 'd':
                printf_int(va_arg(ap, int));
                format++;
                break;
            case 'f':
                printf_double(va_arg(ap, double));
                format++;
                break;
        }
    }

    va_end(ap);
}

int main(void){
    myprintf("wwww%daaaa%faa", 23, -0.14);
}
```
